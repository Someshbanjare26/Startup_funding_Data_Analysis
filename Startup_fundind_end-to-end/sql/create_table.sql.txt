CREATE DATABASE startup_funding_db;
USE startup_funding_db;

CREATE TABLE startup_funding (
    startup_id INT AUTO_INCREMENT PRIMARY KEY,
    startup_name VARCHAR(255),
    industry VARCHAR(100),
    sub_industry VARCHAR(100),
    city VARCHAR(100),
    investors TEXT,
    funding_round VARCHAR(50),
    funding_amount_usd BIGINT,
    funding_date DATE,
    year INT,
    data_source VARCHAR(100)
);
